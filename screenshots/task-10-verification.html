📝 Usage Example

from src.services import create_embeddings

# Create embeddings using factory function
embeddings = create_embeddings(
    provider='openai',
    api_key='sk-your-api-key-here',
    model_name='text-embedding-ada-002'
)

# Generate embedding for a single query
query = "What is the best skill tree for a Witch in Path of Exile?"
embedding = embeddings.embed_query(query)

# Generate embeddings for multiple documents
documents = [
    "The Witch is an intelligence-based caster.",
    "Fireball deals fire damage in an area.",
    "Energy Shield protects against damage."
]
embeddings_list = embeddings.embed_documents(documents)

# Check health
health = embeddings.health_check()
print(health)  # {'status': 'ready', ...}
